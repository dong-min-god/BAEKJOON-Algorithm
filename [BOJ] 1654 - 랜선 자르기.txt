#include <iostream>

using namespace std;

int main()
{
	int K, N;
	int j = 1;
	int num[100001];
	int count = 0;
	cin >> K >> N;
	for (int i = 0; i < K; i++) {
		cin >> num[i];
	}
	while (count != N) {
		count = 0;
		for (int i = 0; i < K; i++) {
			count += num[i] / j;
		}
		j++;
	}
	cout << j << endl;
}
